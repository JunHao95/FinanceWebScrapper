**Step 2: Create tests/test_e2e_golden_path.py:**

```python
import pytest

@pytest.mark.e2e
def test_golden_path_all_tabs_render(page, flask_server):
    """
    Golden path E2E test (all external APIs mocked):
    1. Navigate to app
    2. Enter ticker 'AAPL'
    3. Click Run Analysis
    4. Wait for stock results to appear
    5. Click each of the 4 tabs
    6. Verify each tab has non-empty content
    7. Assert no JS console errors occurred
    """
    js_errors = []
    page.on('pageerror', lambda err: js_errors.append(f'PAGE ERROR: {err}'))
    page.on('console', lambda msg: js_errors.append(f'CONSOLE ERROR: {msg.text}')
            if msg.type == 'error' else None)

    page.goto(flask_server)

    # Fill ticker and submit
    page.fill('{TICKER_SELECTOR}', 'AAPL')
    page.click('{RUN_ANALYSIS_SELECTOR}')

    # Wait for results — stock card should appear (15s timeout — mocked data is fast)
    page.wait_for_selector('{STOCKS_CONTENT_SELECTOR}', state='visible', timeout=15000)

    # Verify each tab renders populated content
    tab_configs = [
        ('{STOCKS_TAB_BUTTON}', '{STOCKS_CONTENT_SELECTOR}'),
        ('{ANALYTICS_TAB_BUTTON}', '{ANALYTICS_CONTENT_SELECTOR}'),
        ('{STOCHASTIC_TAB_BUTTON}', '{STOCHASTIC_CONTENT_SELECTOR}'),
        ('{TRADING_TAB_BUTTON}', '{TRADING_CONTENT_SELECTOR}'),
    ]

    for tab_btn, content_sel in tab_configs:
        page.click(tab_btn)
        page.wait_for_selector(content_sel, state='visible', timeout=10000)
        content = page.locator(content_sel).inner_text()
        assert len(content.strip()) > 0, f'{content_sel} is empty after clicking {tab_btn}'

    # No JS errors
    assert js_errors == [], f'JavaScript errors detected:\n' + '\n'.join(js_errors)
```

Replace all `{SELECTOR}` placeholders with actual selectors found in index.html.

- The flask_server fixture from conftest.py patches all external dependencies before starting the server thread.
- The test needs Playwright/Chromium installed (`playwright install chromium` was done in Plan 01).
- If the Trading Indicators tab requires a separate action to load (e.g., clicking a lookback dropdown), include that step.
